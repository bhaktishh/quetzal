data Action = Action
  { actionName :: String,
    actionRetTy :: (PTy, Maybe String),
    actionStTrans :: (PTm, PTm),
    actionFunc :: Func
  }

data FSM = FSM
  { resourceTy :: PTy,
    resource :: AnnParam,
    stateTy :: PTy,
    initCons :: List Func,
    actions :: List Action
  }

data DirectiveSub = DFSM PTy PTy -- FSM(Store, Access) = DFSM Store Access

data DirectiveTy
  = DAction
      { directiveReturns :: (PTy, Maybe String), -- eg returns LoginResult r
        directiveStTrans :: (PTm, PTm)
      }
  | DInit
  | DRun
      { directiveReturns :: (PTy, Maybe String),
        directiveWith :: (String, PTm), -- eg with this = mkStore("secret", "pub")
        directiveStTrans :: (PTm, PTm)
      }

data Directive = Directive DirectiveSub DirectiveTy

data IFSM = IFSM

data ITmDo = ITmDoLet String (Maybe ITy) ITm
            | ITmDoBind (List String) ITm -- (a,b..) <- tm
            | ITmDoCase (List ITm) (List (List ITm, ITm))
            | ITmDoPure ITm
            | ITmDoIf ITm ITm ITm
            | ITmDoIO ITm

data IFunc = IFunc
  { iFuncName :: String,
    iFuncArgs :: List IAnnParam,
    iFuncBody :: List (ITm, ITm),
    iFuncRetTy :: ITy,
    iWhere :: List ITm
  }

For the FSM construct, from PFSM to IFSM: 
-- idxm: construct an indexed monad type with a constructor for each action specified. first letter of the action constructor is capitalised. needs the actions themselves. additionally, `bind`, `pure`, and `lift` constructors needed -- names should be unique in order to facilitate use with other FSMs. `
-- conc: 

-- currently: action funcs are created and returned as part of `IFSM`. ideally we want them returned simply as regular functions with the transformations applied.

Main.hs

parse -> pToI -> unparse

----------
Parse.hs 
---------
> Full program gets parsed.
> `Func` elements are passed to `mkFSMs`, where `action` and `init` directives are added to an `FSM` construct. In the case of the `run` directive, if the `DirectiveSub` does not exist in the collected map so far, it is inserted as an FSM with no initCons or actions. When the FSM is already in the map, the `action` and `init` directives simply update the right entry, and `run` does nothing. 
> In `mkFSMs`, the first letter of the `actionName` is capitalised, for use as a constructor. 
> FSMs created by `mkFSMs` are added to the final program as `ProgEl`s. This along with the map from `DirectiveSub` to `FSM` is returned to be used by `PToI.hs`. 

--------
PToI.hs
--------
> when translating functions in `trFunc`, we check for a directive
> trFunc takes the FSM map, the original function, and a map of arguments, and returns an IFunc. 
> in the case of FSM run functions, the output function is a wrapper that initialises the concrete resource and calls the monadic run function, which is defined in the `where` block of the wrapper function. 
> the creation of the inner function is like the creation of a monadic block for IO, so they're abstracted into the same function

    -- first create idxm : ITyDecl
    ITyDecl
    { iTyDeclName :: String,
        iTyDeclParams :: List IAnnParam,
        iTyDeclConstructors :: List IConstructor
    }

    iTyDeclName = idxmName is of the form "Idxm<resourceTy>_<stateTy>"
    iTyDeclParams = (t : ITy) -> iStateTy -> (ty -> iStateTy) ( -> ITy, which is implicit)
    iTyDeclConstructors = make constructors for each action, then add pure, lift, bind. 
        > pure and constructors will be named in the form "Pure<resourceTy>_<stateTy>", "Lift<resourceTy>_<stateTy>" to ensure that they do not clash for different FSMs. 
        > bind will always be defined as (>>=) (infix) to enable idris to utilise do notation

    -- now create conc: the concrete resource type, so the elaboration can pass it into and return it from all action functions.
    conc : ITy = trTy resourceTy

    -- finally create run : IFunc 
    IFunc { 
        iFuncName :: String,
        iFuncArgs :: List IAnnParam,
        iFuncBody :: List (ITm, ITm),
        iFuncRetTy :: ITy,
        iWhere :: List ITm 
    }

    iFuncName = of the form "run<resourceTy>_<stateTy>"
    iFuncArgs = [ resourceTy , (<idxmName> ty st1 st2) ]
    iFuncRetTy = IO (t, resourceTy)
    iWhere = [] 
    iFuncBody = a list of LHS and RHS top level pattern match

        > iFuncBody creates one case for each constructor in the type declaration, including pure run and bind
            > for actions, the action constructor's first character is converted to lowercase to obtain the name of the concrete action function

trFunc :: M.Map DirectiveSub FSM -> Func -> M.Map String PTy -> IFunc 
trFunc takes in:
    - the FSM map from DirectiveSub
    - the function to transform 
    - an initially empty map of variables to types, collected as we go down the function
returns an idris function 
IFunc
  { iFuncName :: String,
    iFuncArgs :: List IAnnParam,
    iFuncBody :: List (ITm, ITm),
    iFuncRetTy :: ITy,
    iWhere :: List ITm
  }

-- when a function has a run directive:
    - lookup function directiveSub in the fsm map 
    - if it does not exist, error : no fsm was created with the directive (shoud not happen)
    - if it exists as fsm: (we need to create a function in IO that inits and then runs a function in the idxm, where the latter function is in the where block of the former)
        - translate the function args as args' : [IAnnParam]
        - create main function lhs as funcName applied to args as vars (outerLHS)
        - create main function rhs as mkFSMExecOuterBody funcName directiveWith "<resourceTy>_<stateTy>"(outerRHS)
            ----
            mkFSMExecOuterBody
            ----
            mkFSMExecOuterBody :: String -> (String, PTm) -> String -> ITm 
            mkFSMExecOuterBody takes in: 
                - the name of the outer function (f)
                - the init variable and term from the function directive (initVar. initCall)
                - the string of the form "<resourceTy>_<stateTy>" to call the right functions (str)
            - creates the names of the run function ("run<str>"), result variables ("resVal", "resConc"), and inner function ("<f>'") (runFunc, resVal, resConc, f')
            returns a do block: 
                - let initVar = (tr) initCall 
                - (resVal, resConc) <- runFunc initVar f'
                - pure resVal
            ----
    - make return type : IO (tr) funcRetTy (iFuncRetTy)
    **outer function MUST always be IO**: even if no user annotations are specified, create do block. 

    - make inner func f' : IFunc 
    -- body is the given func body transformed with do notation for the idxm using trMonadListStmt 
        - update the variable map with the init var 

    returns IFunc {
        iFuncName = funcName, 
        iFuncArgs = args', 
        iFuncBody = [outerLHS, outerRHS], 
        iFuncRetTy, 
        iWhere = [ITmFunc f']
    }

trMonadListStmt 
---------------
trMonadListStmt :: Maybe String -> M.Map String PTy -> List Stmt -> List ITmDo
takes in: 
    - Maybe String: if Nothing, the monad is the IO monad; if Just str, this is the fsm variable name (eg "this" but not necessary)
    - M.Map String PTy: a map from variable names to types of vars in context 
    - List Stmt: the actual body to transform 
returns: 
    - (List ITmDo, List ITm)
    where List ITmDo = transformed body in do notation
          List ITm = where block 

case match on 
  StDeclAssign (Maybe PTy) String PTm
  > if rhs is PTmDot: 
    > if mstr is Nothing (monad=IO) and PTmDot subject is IO: 
        > T lhs = IO.f(args) --> {do ..} lhs <- functionCall f args 
    > else if mstr is (Just "this") (monad=FSM, var="this)
        > T lhs = this.f(args) --> {do ..} (lhs, this) <- functionCall f (this : args)
    > else if mstr is (Just str) (monad=FSM, var=str): 
        >  T lhs = a.f(args) --> {do ..} (lhs, a) <- functionCall f (this : args) 
    > else error (only TL matches)
  > else tm = ITmDoLet

  | StAssign String PTm
  > almost same as StDeclAssign except we make sure the assignment is in the kv store

  | StWhile
      { condition :: PTm,
        body :: Stmt
      }
  | StEWhile
      { condition :: PTm,
        body :: Stmt
      }

  | StReturn PTm
  > can only be last statement in block

  | StIf PTm Stmt Stmt
  > append rest of (List Stmt) to each branch, translate, and return ITmDoIf with do blocks wrapping each branch 

  | StEIf PTm Stmt Stmt
  > convert if to pattern match, and then do same as StIf

  | StBlock (List Stmt)
  > no block inside block: only in specific constructs

  | StSwitch Switch
  
  | StSkip
  > do rest 

  | StDot PTm PTm (List PTm)
  > for x.f(args) --> (_, x) <- f args 
  > only applies to fsms

  | StIODot PTm (List PTm)
  > IO.f(args) --> _ <- f(args)
